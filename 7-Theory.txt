6. private 멤버인 id와 name에 접근할 수 없음.
class Student;
class Professor;
class Student {
	int id;
public:
	Student(int id) { this->id = id; }
	friend void show(Student s, Professor p);
}
class Professor {
	string name;
public:
	Professor(string name) { this->name = name; }
	friend void show(Student s, Professor p);
}
void show(Student s, Professor p) {
	cout << s.id << p.name;
}
7. private 멤버인 price에 접근할 수 없음.
class Food {
	int price;
	string name;
public:
	Food(string name, int price);
	void buy();
	friend void Person::shopping(Food food);
}
class Person {
	int id;
public:
	void shopping(Food food) {
		if(food.price < 10000)
			food.buy()
	}
	int get() { return id; }
}
